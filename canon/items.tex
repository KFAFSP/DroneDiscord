\section{Weapons}

\subsection{Melee}

% Recommendations.
\begin{recommendations}
    \recommendation{} Since melee weapons do not have a hit test and always cost \SI{1}{\actionpoints{}}, their damage value must be chosen carefully. Essentially, they convert \AP{} to \termref{damage}[Trigger!Damage].
\end{recommendations}
